💻 实训：银行排队叫号系统

本实训通过实现一个银行排队叫号系统，学习队列数据结构的实际应用。程序完全模拟人群排队全过程：取票进队、排队等待、叫号服务。

1

创建队列接口 IQueue

定义队列的抽象数据类型

▼

队列接口定义了队列的基本操作方法，所有队列实现类都要实现这个接口。

/** * 队列接口 * 定义队列的基本操作 */ public interface IQueue<E> { /** * 入队列操作 * @param item 要入队的元素 * @return 入队是否成功 */ public boolean enqueue(E item); /** * 出队列操作 * @return 队头元素，如果队列为空则返回null */ public E dequeue(); /** * 取队头元素（不删除） * @return 队头元素，如果队列为空则返回null */ public E peek(); /** * 求队列的长度 * @return 队列中元素的个数 */ public int size(); /** * 清空队列 */ public void clear(); /** * 判断队列是否为空 * @return 是否为空 */ public boolean isEmpty(); /** * 判断队列是否为满 * @return 是否为满 */ public boolean isFull(); }
说明：这是一个泛型接口，E表示队列中元素的类型。所有队列实现类（如SeqQueue、LinkQueue）都要实现这些方法。

实现循环顺序队列 SeqQueue

用数组实现循环队列，解决假溢出问题

▼

循环顺序队列使用数组存储数据，通过取模运算实现首尾相接的循环结构，避免假溢出。

/** * 循环顺序队列实现类 * 使用数组存储，通过取模运算实现循环 */ public class SeqQueue<E> implements IQueue<E> { // 队列的容量 private int maxsize; // 存储循环顺序队列中的数据元素 private E[] data; // 指示最近一个已经离开队列的元素所占的位置 private int front; // 指示最近一个进入队列的元素的位置 private int rear; /** * 初始化队列 * @param type 泛型类型 * @param maxsize 队列的容量 */ public SeqQueue(Class<E> type, int maxsize) { data = (E[]) java.lang.reflect.Array.newInstance(type, maxsize); this.maxsize = maxsize; front = rear = -1; } /** * 判断队列是否为空 * 队空条件：front == rear */ public boolean isEmpty() { if (front == rear) { return true; } else { return false; } } /** * 判断循环顺序队列是否为满 * 队满条件：(rear + 1) % maxsize == front */ public boolean isFull() { if ((front == -1 && rear == maxsize - 1) || (rear + 1) % maxsize == front) { return true; } else { return false; } } /** * 入队列操作 * 将rear向后移动一位（取模实现循环），然后存入数据 */ public boolean enqueue(E item) { if (!isFull()) { rear = (rear + 1) % maxsize; data[rear] = item; return true; } else { return false; } } /** * 出队列操作 * 将front向后移动一位（取模实现循环），然后返回数据 */ public E dequeue() { if (!isEmpty()) { front = (front + 1) % maxsize; return data[front]; } else { return null; } } /** * 取队头元素（不删除） */ public E peek() { if (!isEmpty()) { return data[(front + 1) % maxsize]; } else { return null; } } /** * 求队列的长度 * 公式：(rear - front + maxsize) % maxsize */ public int size() { return (rear - front + maxsize) % maxsize; } /** * 清空队列 */ public void clear() { front = rear = -1; } }
循环队列关键公式：
• 队空条件：front == rear
• 队满条件：(rear + 1) % maxsize == front
• 队列长度：(rear - front + maxsize) % maxsize

实现链队列 LinkQueue

用链表实现队列，动态扩展无溢出

▼

链队列使用单链表存储数据，通过front和rear两个指针指示队头和队尾。入队在队尾进行，出队在队头进行。

QueueNode 结点类：

/** * 链队列结点类 * 包含数据域和引用域 */ class QueueNode<E> { E data; // 数据域 QueueNode<E> next; // 引用域（指向下一个结点） /** * 构造函数 */ public QueueNode() { this.data = null; this.next = null; } /** * 带参构造函数 * @param data 结点数据 */ public QueueNode(E data) { this.data = data; this.next = null; } // getter和setter方法 public E getData() { return data; } public void setData(E data) { this.data = data; } public QueueNode<E> getNext() { return next; } public void setNext(QueueNode<E> next) { this.next = next; } }

LinkQueue 链队列类：

/** * 链队列实现类 * 使用单链表存储，front指向队头，rear指向队尾 */ public class LinkQueue<E> implements IQueue<E> { // 队列头指示器 private QueueNode<E> front; // 队列尾指示器 private QueueNode<E> rear; // 队列的容量，假设为0，不限容量 private int maxsize; // 队列数据元素个数 private int size; /** * 初始化链队列（不限容量） */ public LinkQueue() { front = rear = null; size = 0; maxsize = 0; } /** * 初始化限容量的链队列 * @param maxsize 容量限制 */ public LinkQueue(int maxsize) { super(); this.maxsize = maxsize; } /** * 判断队列是否为空 */ public boolean isEmpty() { if ((front == rear) && (size == 0)) { return true; } else { return false; } } /** * 判断队列是否为满 * 如果maxsize为0，说明不限容量 */ public boolean isFull() { if (maxsize != 0 && size == maxsize) { return true; } else { return false; } } /** * 入队列操作 * 在队尾添加新结点 */ public boolean enqueue(E item) { QueueNode<E> newnode = new QueueNode<>(item); if (!isFull()) { if (isEmpty()) { front = newnode; rear = newnode; } else { rear.setNext(newnode); rear = newnode; } ++size; return true; } else { return false; } } /** * 出队列操作 * 从队头删除结点 */ public E dequeue() { if (isEmpty()) { return null; } QueueNode<E> node = front; front = front.getNext(); if (front == null) { rear = null; } --size; return node.getData(); } /** * 取队头元素（不删除） */ public E peek() { if (!isEmpty()) { return front.getData(); } else { return null; } } /** * 求队列的长度 */ public int size() { return size; } /** * 清空队列 */ public void clear() { front = rear = null; size = 0; } }

创建排队机类 QueueMachine

处理顾客取号，分配服务号码

▼

排队机类负责创建队列、处理顾客取号请求。每个顾客按回车键即可获得一个服务号码。

/** * 排队机类 * 该类根据银行工作的开始时间和结束时间创建一个队列， * 当顾客按Enter键时，为顾客获得一个服务号，并将顾客服务号加入队列 */ public class QueueMachine implements Runnable { // 排队机当前最新号码 private int number; // 排队机维持的队列 private IQueue<Integer> queue; // 排队机工作开始时间，例如：0800为早上8点 private String starttime; // 排队机工作结束时间，例如：1630为下午4点半 private String endtime; /** * 构造函数 * @param starttime 工作开始时间 * @param endtime 工作结束时间 */ public QueueMachine(String starttime, String endtime) { queue = new SeqQueue<Integer>(Integer.class, 100); this.starttime = starttime; this.endtime = endtime; } /** * 获取服务队列 * @return 队列 */ public IQueue<Integer> getQueue() { return queue; } /** * 排队机工作线程 * 等待顾客按回车键取号 */ public void run() { java.util.Scanner sc = new java.util.Scanner(System.in); java.text.SimpleDateFormat df = new java.text.SimpleDateFormat("HHmm"); while (true) { System.out.println("按回车键获取号码:"); sc.nextLine(); String time = df.format(new java.util.Date()); if (!(time.compareTo(starttime) >= 0 && time.compareTo(endtime) <= 0)) { break; } int callnumber = ++number; if (queue.enqueue(callnumber)) { System.out.println(String.format("您的号码是:%d, 你前面有 %d 位, 请等待!", callnumber, queue.size() - 1)); } else { System.out.println("现在业务繁忙, 请稍后再试!"); number--; } } sc.close(); System.out.println("已到下班时间，请明天再来"); } }

创建服务窗口类 ServiceWindow

从队列取顾客进行服务

▼

服务窗口类是一个线程，每隔10秒从队列中取出一个顾客进行服务。使用synchronized保证线程安全。

/** * 服务窗口类 * 服务窗口的职能是为排队的服务，每当服务窗口按照先进先出的原则 * 从队列中选取一个人进行服务时，就有一个顾客出队 */ public class ServiceWindow implements Runnable { // 服务队列 private IQueue<Integer> queue; /** * 在构造函数中指定服务的队列 * @param queue 服务队列 */ public ServiceWindow(IQueue<Integer> queue) { this.queue = queue; } /** * 服务窗口工作线程 * 每10秒叫号一次 */ public void run() { while (true) { synchronized (queue) { if (queue.size() > 0) { System.out.println(String.format("请 %d 号到 %s 号窗口!", queue.dequeue(), Thread.currentThread().getName())); } } try { Thread.sleep(10000); // 模拟服务时间10秒 } catch (InterruptedException e) { System.out.println(e.getMessage()); } } } }
多线程要点：
• synchronized(queue)：保证多线程安全访问共享队列
• Thread.currentThread().getName()：获取当前线程名称（窗口号）
• Thread.sleep(10000)：暂停10秒，模拟服务时间
