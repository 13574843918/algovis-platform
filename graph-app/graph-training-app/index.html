import java.util.*;

/**
 * 邻接矩阵实现图结构
 * 使用一维数组 vexs[] 存储顶点，二维数组 edges[][] 存储边
 */
public class GraphAdjMatrix<E> implements IGraph<E> {
    private E[] vexs;           // 顶点数组
    private int[][] edges;      // 邻接矩阵
    private int numOfVexs;      // 顶点实际数量
    private int maxNumOfVexs;   // 顶点最大数量
    private boolean[] visited;  // 访问标记

    // ---- 构造函数 ----
    public GraphAdjMatrix(int maxNumOfVexs, Class<E> type) {
        this.maxNumOfVexs = maxNumOfVexs;
        vexs = (E[]) Array.newInstance(type, maxNumOfVexs);
        edges = new int[maxNumOfVexs][maxNumOfVexs];
        numOfVexs = 0;
    }

    // ---- 获取顶点数量 ----
    public int getNumOfVertex() {
        return numOfVexs;
    }

    // ---- 插入顶点 ----
    public boolean insertVex(E v) {
        if (numOfVexs >= maxNumOfVexs)
            return false;
        vexs[numOfVexs++] = v;
        return true;
    }

    // ---- 插入边 ----
    public boolean insertEdge(int v1, int v2, int weight) {
        if (v1 < 0 || v2 < 0 || v1 >= numOfVexs || v2 >= numOfVexs)
            throw new ArrayIndexOutOfBoundsException();
        edges[v1][v2] = weight;
        edges[v2][v1] = weight;  // 无向图：对称赋值
        return true;
    }

    // ---- 删除顶点 ----
    public boolean deleteVex(E v) {
        for (int i = 0; i < numOfVexs; i++) {
            if (vexs[i].equals(v)) {
                // 删除顶点：移动顶点数组（后面的前移）
                for (int j = i; j < numOfVexs - 1; j++) {
                    vexs[j] = vexs[j + 1];
                }
                vexs[numOfVexs - 1] = null;
                // 删除顶点：移动邻接矩阵的行
                for (int row = i; row < numOfVexs - 1; row++) {
                    for (int col = 0; col < numOfVexs; col++) {
                        edges[row][col] = edges[row + 1][col];
                    }
                }
                // 删除顶点：移动邻接矩阵的列
                for (int col = i; col < numOfVexs - 1; col++) {
                    for (int row = 0; row < numOfVexs; row++) {
                        edges[row][col] = edges[row][col + 1];
                    }
                }
                numOfVexs--;
                return true;
            }
        }
        return false;
    }

    // ---- 删除边 ----
    public boolean deleteEdge(int v1, int v2) {
        if (v1 < 0 || v2 < 0 || v1 >= numOfVexs || v2 >= numOfVexs)
            throw new ArrayIndexOutOfBoundsException();
        edges[v1][v2] = 0;
        edges[v2][v1] = 0;
        return true;
    }

    // ---- 定位顶点（返回索引）----
    public int indexOfVex(E v) {
        for (int i = 0; i < numOfVexs; i++) {
            if (vexs[i].equals(v)) {
                return i;
            }
        }
        return -1;
    }

    // ---- 查找顶点（返回值）----
    public E valueOfVex(int v) {
        if (v < 0 || v >= numOfVexs)
            return null;
        return vexs[v];
    }

    // ---- 查找边（返回权值）----
    public int getEdge(int v1, int v2) {
        if (v1 < 0 || v2 < 0 || v1 >= numOfVexs || v2 >= numOfVexs)
            throw new ArrayIndexOutOfBoundsException();
        return edges[v1][v2];
    }
}
✅ 关键理解：邻接矩阵用二维数组表示图的边，edges[i][j] 表示顶点 i 到 j 的边权值。空间复杂度 O(V²)，适合稠密图。插入/删除顶点需要移动矩阵，时间复杂度 O(V²)；插入/删除边只需 O(1)。

    // ========== 遍历算法 ==========

    // ---- 深度优先遍历（DFS，邻接矩阵版）----
    public String depthFirstSearch(int v) {
        if (v < 0 || v >= numOfVexs)
            throw new ArrayIndexOutOfBoundsException();
        visited = new boolean[numOfVexs];
        StringBuilder sb = new StringBuilder();
        DFS(v, visited, sb);
        return sb.length() > 0 ? sb.substring(0, sb.length() - 1) : "";
    }

    private String DFS(int v, boolean[] visited, StringBuilder sb) {
        visited[v] = true;
        sb.append(vexs[v] + ",");
        for (int i = 0; i < numOfVexs; i++) {
            if ((edges[v][i] != 0 && edges[v][i] != Integer.MAX_VALUE)
                && !visited[i]) {
                DFS(i, visited, sb);
            }
        }
        return sb.toString();
    }

    // ---- 广度优先遍历（BFS，邻接矩阵版）----
    public String breadthFirstSearch(int v) {
        if (v < 0 || v >= numOfVexs)
            throw new ArrayIndexOutOfBoundsException();
        visited = new boolean[numOfVexs];
        StringBuilder sb = new StringBuilder();
        Queue<Integer> queue = new LinkedList<Integer>();
        queue.offer(v);
        visited[v] = true;
        while (!queue.isEmpty()) {
            v = queue.poll();
            sb.append(vexs[v] + ",");
            for (int i = 0; i < numOfVexs; i++) {
                if ((edges[v][i] != 0 && edges[v][i] != Integer.MAX_VALUE)
                     && !visited[i]) {
                    queue.offer(i);
                    visited[i] = true;
                }
            }
        }
        return sb.length() > 0 ? sb.substring(0, sb.length() - 1) : null;
    }
}

import java.util.*;

/**
 * 邻接表实现图结构
 * 使用顶点数组 vexs[] + 边链表存储
 */
public class GraphAdjList<E> implements IGraph<E> {

    // ---- 邻接表中边信息 ----
    private class ENode {
        int adjvex;      // 邻接顶点序号
        int weight;      // 边的权值
        ENode nextadj;   // 下一个邻接表结点
        public ENode(int adjvex, int weight) {
            this.adjvex = adjvex;
            this.weight = weight;
        }
    }

    // ---- 邻接表中顶点信息 ----
    private class VNode {
        E data;           // 顶点信息
        ENode firstadj;   // 相邻的所有结点组成的单链表的头指针
    }

    private VNode[] vexs;        // 顶点数组
    private int numOfVexs;       // 顶点的实际数量
    private int maxNumOfVexs;    // 顶点的最大数量
    private boolean[] visited;   // 判断顶点是否被访问过

    // ---- 构造函数 ----
    public GraphAdjList(int maxNumOfVexs) {
        this.maxNumOfVexs = maxNumOfVexs;
        vexs = (VNode[]) Array.newInstance(VNode.class, maxNumOfVexs);
        numOfVexs = 0;
    }

    // ---- 获取顶点数量 ----
    public int getNumOfVertex() { return numOfVexs; }

    // ---- 插入顶点 ----
    public boolean insertVex(E v) {
        if (numOfVexs >= maxNumOfVexs) return false;
        VNode vex = new VNode();
        vex.data = v;
        vexs[numOfVexs++] = vex;
        return true;
    }

    // ---- 插入边（头插法）----
    public boolean insertEdge(int v1, int v2, int weight) {
        if (v1 < 0 || v2 < 0 || v1 >= numOfVexs || v2 >= numOfVexs)
            throw new ArrayIndexOutOfBoundsException();
        // v1→v2 的边
        ENode vex1 = new ENode(v2, weight);
        if (vexs[v1].firstadj == null) {
            vexs[v1].firstadj = vex1;
        } else {
            vex1.nextadj = vexs[v1].firstadj;
            vexs[v1].firstadj = vex1;
        }
        // v2→v1 的边（无向图）
        ENode vex2 = new ENode(v1, weight);
        if (vexs[v2].firstadj == null) {
            vexs[v2].firstadj = vex2;
        } else {
            vex2.nextadj = vexs[v2].firstadj;
            vexs[v2].firstadj = vex2;
        }
        return true;
    }

    // ---- 删除顶点（邻接表）----
    public boolean deleteVex(E v) {
        for (int i = 0; i < numOfVexs; i++) {
            if (vexs[i].data.equals(v)) {
                // 移动顶点数组
                for (int j = i; j < numOfVexs - 1; j++) {
                    vexs[j] = vexs[j + 1];
                }
                vexs[numOfVexs - 1] = null;
                numOfVexs--;
                // 删除所有指向该顶点的边，并更新邻接点下标
                ENode current;
                ENode previous;
                for (int j = 0; j < numOfVexs; j++) {
                    if (vexs[j].firstadj == null) continue;
                    // 检查头结点
                    if (vexs[j].firstadj.adjvex == i) {
                        vexs[j].firstadj = vexs[j].firstadj.nextadj;
                        continue;
                    }
                    // 检查链表中的其他结点
                    current = vexs[j].firstadj;
                    while (current != null) {
                        previous = current;
                        current = current.nextadj;
                        if (current != null && current.adjvex == i) {
                            previous.nextadj = current.nextadj;
                            break;
                        }
                    }
                }
                // 更新剩余顶点的邻接点下标（大于i的减1）
                for (int j = 0; j < numOfVexs; j++) {
                    current = vexs[j].firstadj;
                    while (current != null) {
                        if (current.adjvex > i)
                            current.adjvex--;
                        current = current.nextadj;
                    }
                }
                return true;
            }
        }
        return false;
    }

    // ---- 删除边 ----
    public boolean deleteEdge(int v1, int v2) {
        if (v1 < 0 || v2 < 0 || v1 >= numOfVexs || v2 >= numOfVexs)
            throw new ArrayIndexOutOfBoundsException();
        // 删除 v1→v2 的边
        ENode current = vexs[v1].firstadj;
        ENode previous = null;
        while (current != null) {
            if (current.adjvex == v2) {
                if (previous == null) {
                    vexs[v1].firstadj = current.nextadj;
                } else {
                    previous.nextadj = current.nextadj;
                }
                break;
            }
            previous = current;
            current = current.nextadj;
        }
        // 删除 v2→v1 的边
        current = vexs[v2].firstadj;
        previous = null;
        while (current != null) {
            if (current.adjvex == v1) {
                if (previous == null) {
                    vexs[v2].firstadj = current.nextadj;
                } else {
                    previous.nextadj = current.nextadj;
                }
                break;
            }
            previous = current;
            current = current.nextadj;
        }
        return true;
    }

    // ---- 定位顶点 ----
    public int indexOfVex(E v) {
        for (int i = 0; i < numOfVexs; i++) {
            if (vexs[i].data.equals(v)) return i;
        }
        return -1;
    }

    // ---- 查找顶点 ----
    public E valueOfVex(int v) {
        if (v < 0 || v >= numOfVexs) return null;
        return vexs[v].data;
    }

    // ---- 查找边（返回权值）----
    public int getEdge(int v1, int v2) {
        if (v1 < 0 || v2 < 0 || v1 >= numOfVexs || v2 >= numOfVexs)
            throw new ArrayIndexOutOfBoundsException();
        ENode current = vexs[v1].firstadj;
        while (current != null) {
            if (current.adjvex == v2) {
                return current.weight;
            }
            current = current.nextadj;
        }
        return 0;
    }

    // ---- 深度优先遍历（邻接表版）----
    public String depthFirstSearch(int v) {
        if (v < 0 || v >= numOfVexs)
            throw new ArrayIndexOutOfBoundsException();
        visited = new boolean[numOfVexs];
        StringBuilder sb = new StringBuilder();
        DFS(v, visited, sb);
        return sb.length() > 0 ? sb.substring(0, sb.length() - 1) : "";
    }

    private String DFS(int v, boolean[] visited, StringBuilder sb) {
        visited[v] = true;
        sb.append(vexs[v].data + ",");
        ENode current = vexs[v].firstadj;
        while (current != null) {
            if (!visited[current.adjvex]) {
                DFS(current.adjvex, visited, sb);
            }
            current = current.nextadj;
        }
        return sb.toString();
    }

    // ---- 广度优先遍历（邻接表版）----
    public String breadthFirstSearch(int v) {
        if (v < 0 || v >= numOfVexs)
            throw new ArrayIndexOutOfBoundsException();
        visited = new boolean[numOfVexs];
        StringBuilder sb = new StringBuilder();
        Queue<Integer> queue = new LinkedList<Integer>();
        queue.offer(v);
        visited[v] = true;
        ENode current;
        while (!queue.isEmpty()) {
            v = queue.poll();
            sb.append(vexs[v].data + ",");
            current = vexs[v].firstadj;
            while (current != null) {
                if (!visited[current.adjvex]) {
                    queue.offer(current.adjvex);
                    visited[current.adjvex] = true;
                }
                current = current.nextadj;
            }
        }
        return sb.length() > 0 ? sb.substring(0, sb.length() - 1) : "";
    }

    // ---- Dijkstra最短路径（邻接表版）----
    public int[] dijkstra(int v) {
        if (v < 0 || v >= numOfVexs)
            throw new ArrayIndexOutOfBoundsException();
        boolean[] st = new boolean[numOfVexs];  // 默认初始为 false
        int[] distance = new int[numOfVexs];    // 存放源点到其他点的距离
        // 初始化所有距离为最大值
        for (int i = 0; i < numOfVexs; i++) {
            distance[i] = Integer.MAX_VALUE;
        }
        // 根据邻接表初始化源点直接可达的距离
        ENode current;
        current = vexs[v].firstadj;
        while (current != null) {
            distance[current.adjvex] = current.weight;
            current = current.nextadj;
        }
        distance[v] = 0;
        st[v] = true;
        // 处理从源点到其余顶点的最短路径
        for (int i = 0; i < numOfVexs; i++) {
            int min = Integer.MAX_VALUE;
            int index = -1;
            // 比较从源点到其余顶点的路径长度
            for (int j = 0; j < numOfVexs; j++) {
                // 从源点到 j 顶点的最短路径还没有找到
                if (st[j] == false) {
                    // 从源点到 j 顶点的路径长度最小
                    if (distance[j] < min) {
                        index = j;
                        min = distance[j];
                    }
                }
            }
            // 找到源点到索引为 index 顶点的最短路径长度
            if (index != -1)
                st[index] = true;
            // 更新当前最短路径及距离
            for (int w = 0; w < numOfVexs; w++) {
                if (st[w] == false) {
                    current = vexs[w].firstadj;
                    while (current != null) {
                        if (current.adjvex == index)
                            if ((min + current.weight) < distance[w]) {
                                distance[w] = min + current.weight;
                                break;
                            }
                        current = current.nextadj;
                    }
                }
            }
        }
        return distance;
    }
}

    // ========== Dijkstra 最短路径（邻接矩阵版）==========

    public int[] dijkstra(int v) {
        if (v < 0 || v >= numOfVexs)
            throw new ArrayIndexOutOfBoundsException();
        boolean[] st = new boolean[numOfVexs];
        int[] distance = new int[numOfVexs];
        // 将邻接矩阵中非对角线上为零的边设成整数的最大值
        for (int i = 0; i < numOfVexs; i++) {
            for (int j = i + 1; j < numOfVexs; j++) {
                if (edges[i][j] == 0) {
                    edges[i][j] = Integer.MAX_VALUE;
                    edges[j][i] = Integer.MAX_VALUE;
                }
            }
        }
        // 初始化距离数组
        for (int i = 0; i < numOfVexs; i++)
            distance[i] = edges[v][i];
        st[v] = true;
        // 处理从源点到其余顶点的最短路径
        for (int i = 0; i < numOfVexs; ++i) {
            int min = Integer.MAX_VALUE;
            int index = -1;
            for (int j = 0; j < numOfVexs; ++j) {
                if (!st[j] && distance[j] < min) {
                    index = j;
                    min = distance[j];
                }
            }
            if (index != -1) st[index] = true;
            for (int w = 0; w < numOfVexs; w++) {
                if (st[w] == false) {
                    if (edges[index][w] != Integer.MAX_VALUE
                        && (min + edges[index][w] < distance[w])) {
                        distance[w] = min + edges[index][w];
                    }
                }
            }
        }
        return distance;
    }
}

✅ 关键理解：Dijkstra 算法是贪心算法，每次选取当前已知最短距离的顶点加入集合 S，然后更新其他顶点的距离。时间复杂度 O(V²)，可用最小堆优化到 O((V+E)logV)。注意：邻接矩阵版会修改原始 edges 数组（将0替换为 MAX_VALUE），实际开发中应先备份原数据。
