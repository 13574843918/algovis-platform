任务1：链式通用二叉树类 LinkBiTree（遍历）

用链式存储实现通用二叉树，支持 4 种遍历方式

▼
📋 实训目标：理解二叉树链式存储结构（结点 = 数据 + 左孩子 + 右孩子），掌握先序（DLR）、中序（LDR）、后序（LRD）、层次遍历 4 种遍历算法的实现思想与访问顺序差异。

链式二叉树用 Node 结点存储，每个结点包含数据域、左孩子指针、右孩子指针：

A / \ B C / \ \ D E F 先序(DLR): A B D E C F → 根 → 左 → 右 中序(LDR): D B E A C F → 左 → 根 → 右 后序(LRD): D E B F C A → 左 → 右 → 根 层次: A B C D E F → 按层从上到下
src/btree/Node.java 
/**
 * 二叉树结点类（链式存储）
 */
public class Node<E> {
    private E data;          // 数据域，存储数据元素
    private Node<E> lchild;  // 左孩子指针
    private Node<E> rchild;  // 右孩子指针

    public Node(E val) {
        this.data = val;
        this.lchild = null;
        this.rchild = null;
    }

    public Node(E val, Node<E> l, Node<E> r) {
        this.data = val;
        this.lchild = l;
        this.rchild = r;
    }

    public E getData() { return data; }
    public void setData(E val) { data = val; }
    public Node<E> getLchild() { return lchild; }
    public void setLchild(Node<E> l) { lchild = l; }
    public Node<E> getRchild() { return rchild; }
    public void setRchild(Node<E> r) { rchild = r; }
}
src/btree/LinkBiTree.java 
import java.util.*;

/**
 * 链式存储通用二叉树类
 * 支持：创建、遍历、插入、删除、查找等基本操作
 */
public class LinkBiTree<E> {
    private Node<E> root;  // 二叉树的根结点

    public LinkBiTree() { root = null; }

    // ---- 初始化二叉树 ----
    public void init(E[] preOrder, E[] inOrder) {
        root = init(preOrder, 0, preOrder.length - 1,
                    inOrder, 0, inOrder.length - 1);
    }

    private Node<E> init(E[] pre, int preL, int preR,
                          E[] in, int inL, int inR) {
        if (preL > preR) { return null; }
        Node<E> p = new Node<E>(pre[preL]);
        int i = 0;
        for (int k = inL; k <= inR; k++) {
            if (in[k].equals(pre[preL])) { i = k; break; }
        }
        p.setLchild(init(pre, preL + 1, preL + i - inL, in, inL, i - 1));
        p.setRchild(init(pre, preL + i - inL + 1, preR, in, i + 1, inR));
        return p;
    }

    // 获取根结点
    public Node<E> getRoot() { return root; }

    // 判断树是否为空
    public boolean isEmpty() { return root == null; }

    // 获取左子树
    public Node<E> getLchild(Node<E> p) { return p.getLchild(); }

    // 获取右子树
    public Node<E> getRchild(Node<E> p) { return p.getRchild(); }

    // 插入数据到左子树
    public void insertL(E val, Node<E> p) {
        Node<E> node = new Node<E>(val);
        node.setLchild(p.getLchild());
        p.setLchild(node);
    }

    // 插入数据到右子树
    public void insertR(E val, Node<E> p) {
        Node<E> node = new Node<E>(val);
        node.setRchild(p.getRchild());
        p.setRchild(node);
    }

    // 删除左子树
    public Node<E> deleteL(Node<E> p) {
        if ((p == null) || (p.getLchild() == null)) { return null; }
        Node<E> tmp = p.getLchild();
        p.setLchild(null);
        return tmp;
    }

    // 删除右子树
    public Node<E> deleteR(Node<E> p) {
        if ((p == null) || (p.getRchild() == null)) { return null; }
        Node<E> tmp = p.getRchild();
        p.setRchild(null);
        return tmp;
    }

    // 先序遍历（DLR）
    public void preOrder(Node<E> p) {
        if (p != null) {
            System.out.print(p.getData() + " ");    // 访问根结点
            preOrder(p.getLchild());                   // 递归先序遍历左子树
            preOrder(p.getRchild());                   // 递归先序遍历右子树
        }
    }

    // 中序遍历（LDR）
    public void inOrder(Node<E> p) {
        if (p != null) {
            inOrder(p.getLchild());                     // 递归中序遍历左子树
            System.out.print(p.getData() + " "); // 访问根结点
            inOrder(p.getRchild());                     // 递归中序遍历右子树
        }
    }

    // 后序遍历（LRD）
    public void postOrder(Node<E> p) {
        if (p != null) {
            postOrder(p.getLchild());                 // 递归后序遍历左子树
            postOrder(p.getRchild());                 // 递归后序遍历右子树
            System.out.print(p.getData() + "  "); // 访问根结点
        }
    }

    // 层次遍历（用队列）
    public void levelOrder(Node<E> p) {
        if (p == null) { return; }
        Queue<Node<E>> q = new LinkedList<Node<E>>();
        q.add(p);
        while (!q.isEmpty()) {
            Node<E> tmp = q.poll();
            System.out.print(tmp.getData() + " ");
            if (tmp.getLchild() != null) { q.add(tmp.getLchild()); }
            if (tmp.getRchild() != null) { q.add(tmp.getRchild()); }
        }
    }
}
✅ 关键理解：先序遍历的访问顺序是"根-左-右"，对应递归实现中"访问"语句位于两个递归调用之前；中序遍历将访问语句放在两次递归调用之间；后序遍历则放在最后。层次遍历使用队列保证按层访问。

import java.util.*;
/**
 * 顺序存储最优二叉树（哈夫曼树）
 * 使用 HData[] 数组存储，通过下标构建父子关系
 */
public class SeqHuffmanTree {
    private HData[] data;   // 存储哈夫曼树的数组
    private int root;     // 根结点下标

    // ---- 初始化树并建树 ----
    public void create(String message) {
        String str = message.trim();
        // 统计字符频次
        Map<Character, Integer> charSet = new HashMap<>();
        for (int i = 0; i < str.length(); i++) {
            char key = str.charAt(i);
            charSet.put(key, charSet.getOrDefault(key, 0) + 1);
        }
        int n = charSet.size();
        data = new HData[4 * n];   // 哈夫曼树最多 2n-1 个结点，开 4n 足够
        List<Map.Entry<Character, Integer>> sorted =
                sortMapByValueKey(charSet);
        // 将字符结点初始化到数组前 n 个位置
        int idx = 0;
        for (Map.Entry<Character, Integer> entry : sorted) {
            char key = entry.getKey();
            int v = entry.getValue();
            data[idx] = new HData(key + "", v);
            idx++;
        }
        // 构建哈夫曼树：每次选取最小的两个结点合并
        int s1, s2;
        for (int i = 0; i < n - 1; i++) {
            select(idx, s1 = -1, s2 = -1);
            data[s2].setWeight(Integer.MAX_VALUE);   // 标记为已选
            select(idx, s1, s2);
            data[s2].setWeight(Integer.MAX_VALUE);
            int sum = data[s1].getWeight() + data[s2].getWeight();
            data[idx] = new HData("总权值", sum);
            data[idx].setLchild(s1);
            data[idx].setRchild(s2);
            idx++;
        }
        root = idx - 1;
    }

    // ---- 辅助：选取权值最小的两个结点 ----
    private void select(int n, int s1, int s2) {
        // s1, s2 返回最小两个结点下标（排除 MAX_VALUE）
    }

    // ---- 遍历并生成哈夫曼编码 ----
    public int traverse() {
        return preOrder(root, "");
    }

    private int preOrder(int i, String code) {
        int sum = 0;
        if (data[i] != null) {
            data[i].setCode(code);
            if (isLeaf(i)) {
                System.out.println(data[i].getName() + ":" + data[i].getCode());
                return data[i].getWeight() * data[i].getCode().length();
            }
            if (data[i].getLchild() >= 0)
                sum += preOrder(data[i].getLchild(), code + "0");
            if (data[i].getRchild() >= 0)
                sum += preOrder(data[i].getRchild(), code + "1");
        }
        return sum;
    }

    private boolean isLeaf(int i) {
        return data[i].getLchild() < 0 && data[i].getRchild() < 0;
    }

    // ---- 获取根结点下标 ----
    public int getRoot() { return root; }

    // ---- 查找结点 ----
    public HData search(int p, String name) {
        if (p < 0 || p >= data.length || data[p] == null) { return null; }
        if (data[p].getName().equals(name)) { return data[p]; }
        HData res = null;
        if (data[p].getLchild() >= 0) { res = search(data[p].getLchild(), name); }
        if (res != null) { return res; }
        if (data[p].getRchild() >= 0) { return search(data[p].getRchild(), name); }
        return null;
    }

    // ---- 文本编码 ----
    public String codes(String message) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < message.length(); i++) {
            char c = message.charAt(i);
            HData hn = search(root, c + "");
            sb.append(hn.getData().getCode());
        }
        return sb.toString();
    }

    // ---- 报文解码（层次遍历）----
    public String decodes(String codes) {
        Queue<Integer> q = new LinkedList<>();
        q.add(root);
        String str = "";
        while (!q.isEmpty()) {
            int tmp = q.poll();
            if (!codes.startsWith(data[tmp].getCode())) continue;
            if (isLeaf(tmp)) {
                str = str + data[tmp].getName();
                codes = codes.substring(data[tmp].getCode().length());
                if (codes.length() > 0) { q.clear(); q.add(root); continue; }
            }
            if (data[tmp].getLchild() >= 0) { q.add(data[tmp].getLchild()); }
            if (data[tmp].getRchild() >= 0) { q.add(data[tmp].getRchild()); }
        }
        return codes.length() > 0 ? "" : str;
    }

    // ---- 辅助：对 Map 按 value 和 key 升序排序 ----
    private List<Map.Entry<Character, Integer>> sortMapByValueKey(
            Map<Character, Integer> map) {
        List<Map.Entry<Character, Integer>> list =
                new ArrayList<>(map.entrySet());
        Collections.sort(list, (a, b) -> {
            int cmp = a.getValue().compareTo(b.getValue());
            if (cmp != 0) return cmp;
            return a.getKey().compareTo(b.getKey());
        });
        return list;
    }
}

任务3：链式存储实现通用二叉树（扩展）

在前序遍历建树的基础上实现二叉搜索树（BST）功能

▼
📋 实训目标：在前两个任务基础上，扩展链式二叉树的功能，增加二叉搜索树（BST）的插入、查找、删除操作，理解链式存储结构在动态操作中的灵活性。
src/btree/LinkBiTreeExt.java 
import java.util.*;

/**
 * 链式通用二叉树扩展类
 * 在 LinkBiTree 基础上增加 BST 功能：插入、查找、删除
 */
public class LinkBiTreeExt<E extends Comparable<E>> {

    // ---- BST 插入（非递归）----
    public Node<E> insert(E val) {
        Node<E> newNode = new Node<E>(val);
        if (root == null) { root = newNode; return root; }
        Node<E> p = root, parent = null;
        while (p != null) {
            parent = p;
            if (val.compareTo(p.getData()) < 0) { p = p.getLchild(); }
            else { p = p.getRchild(); }
        }
        if (val.compareTo(parent.getData()) < 0) { parent.setLchild(newNode); }
        else { parent.setRchild(newNode); }
        return newNode;
    }

    // ---- BST 查找 ----
    public Node<E> search(E val) {
        Node<E> p = root;
        while (p != null) {
            if (val.equals(p.getData())) { return p; }
            if (val.compareTo(p.getData()) < 0) { p = p.getLchild(); }
            else { p = p.getRchild(); }
        }
        return null;
    }

    // ---- BST 删除（分三种情况）----
    public boolean delete(E val) {
        Node<E> p = root, parent = null;
        while (p != null) {            // 查找待删除结点
            if (val.equals(p.getData())) { break; }
            parent = p;
            if (val.compareTo(p.getData()) < 0) { p = p.getLchild(); }
            else { p = p.getRchild(); }
        }
        if (p == null) { return false; }   // 未找到

        // 情况1：叶子结点，直接删除
        if (p.getLchild() == null && p.getRchild() == null) {
            if (parent == null) { root = null; }
            else if (parent.getLchild() == p) { parent.setLchild(null); }
            else { parent.setRchild(null); }
            return true;
        }
        // 情况2：只有右子树，用右子树替代
        if (p.getLchild() == null) {
            if (parent == null) { root = p.getRchild(); }
            else if (parent.getLchild() == p) { parent.setLchild(p.getRchild()); }
            else { parent.setRchild(p.getRchild()); }
            return true;
        }
        // 情况3：只有左子树，用左子树替代
        if (p.getRchild() == null) {
            if (parent == null) { root = p.getLchild(); }
            else if (parent.getLchild() == p) { parent.setLchild(p.getLchild()); }
            else { parent.setRchild(p.getLchild()); }
            return true;
        }
        // 情况4：左右子树都有，用后继结点（中序后继：右子树最左下）替代
        Node<E> successorParent = p, successor = p.getRchild();
        while (successor.getLchild() != null) {
            successorParent = successor;
            successor = successor.getLchild();
        }
        p.setData(successor.getData());   // 用后继值覆盖当前结点
        // 删除后继结点（后继一定没有左子树）
        if (successorParent == p) { successorParent.setRchild(successor.getRchild()); }
        else { successorParent.setLchild(successor.getRchild()); }
        return true;
    }
}
✅ BST 删除三种情况：① 叶子结点 → 直接删除；② 单子树 → 子树提升；③ 双子树 → 用中序后继（右子树最左下）替换后删除后继。哈夫曼树本质上也是一棵二叉树，只不过它的构建规则由权值决定而非比较关系。

 * 链式存储最优二叉树（哈夫曼树）
 * 使用 PriorityQueue（最小堆）+ 链式结点实现
 */
public class LinkedHuffmanTree {
    private Node<HData> root;   // 最优二叉树的根结点

    private PriorityQueue<Node<HData>> queue =
        new PriorityQueue<>(new Comparator<Node<HData>>() {
            public int compare(Node<HData> n1, Node<HData> n2) {
                int x1 = n1.getData().getWeight() - n2.getData().getWeight();
                int x2 = n1.getData().getName().length()
                           - n2.getData().getName().length();
                if (x1 != 0) return x1;
                if (x2 != 0) return x2;
                return n1.getData().getName().compareTo(n2.getData().getName());
            }
        });

    // ---- 构建最优二叉树 ----
    public void create(String message) {
        String str = message.trim();
        System.out.println("待处理的文本长度: " + str.length());
        Map<Character, Integer> charSet = new HashMap<>();
        for (int i = 0; i < str.length(); i++) {
            char key = str.charAt(i);
            charSet.put(key, charSet.getOrDefault(key, 0) + 1);
        }
        int cnt = charSet.size();
        System.out.println("待处理的字符数量: " + cnt);
        System.out.println("各个字符出现频次: ");
        List<Map.Entry<Character, Integer>> sorted =
                sortMapByValueKey(charSet);
        sorted.forEach(item -> {
            char key = item.getKey();
            int v = item.getValue();
            HData hnode = new HData(key + "", v);
            Node<HData> node = new Node<HData>(hnode);
            queue.offer(node);
            System.out.print(key + ":" + v + " ");
        });
        System.out.println();
        cnt = 0;
        while (queue.size() > 1) {
            Node<HData> lchild = queue.poll();
            Node<HData> rchild = queue.poll();
            int newWeight = lchild.getData().getWeight() +
                            rchild.getData().getWeight();
            cnt++;
            HData hn = new HData("M" + cnt, newWeight);
            Node<HData> newNode = new Node<HData>(hn, lchild, rchild);
            queue.offer(newNode);
        }
        root = queue.poll();
    }

    // ---- 判断是否为叶结点 ----
    public boolean isLeaf(Node<HData> p) {
        return p.getLchild() == null && p.getRchild() == null;
    }

    // ---- 获得根结点 ----
    public Node<HData> getRoot() { return root; }

    // ---- 遍历二叉树（先序，生成编码并计算总长度）----
    public int traverse(int i) { return preOrder(root, ""); }

    private int preOrder(Node<HData> p, String code) {
        int sum = 0;
        if (p != null) {
            HData hnode = p.getData();
            hnode.setCode(code);
            if (isLeaf(p)) {
                System.out.println(hnode.getName() + ":" + hnode.getCode());
                return hnode.getWeight() * hnode.getCode().length();
            }
            if (p.getLchild() != null)
                sum += preOrder(p.getLchild(), code + "0");
            if (p.getRchild() != null)
                sum += preOrder(p.getRchild(), code + "1");
        }
        return sum;
    }

    // ---- 查找结点 ----
    public HData search(Node<HData> p, String name) {
        if (p == null) return null;
        if (p.getData().getName().equals(name)) { return p.getData(); }
        HData res = search(p.getLchild(), name);
        if (res != null) { return res; }
        return search(p.getRchild(), name);
    }

    // ---- 文本编码 ----
    public String codes(String message) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < message.length(); i++) {
            char c = message.charAt(i);
            HData hn = search(root, c + "");
            if (hn == null)
                throw new RuntimeException(c + ":未找到该字符的编码!");
            sb.append(hn.getCode());
        }
        return sb.toString();
    }

    // ---- 报文解码（层次遍历）----
    public String decodes(String codes) {
        Queue<Node<HData>> q = new LinkedList<Node<HData>>();
        q.add(root);
        StringBuilder sb = new StringBuilder();
        while (!q.isEmpty()) {
            Node<HData> node = q.poll();
            HData hnode = node.getData();
            if (!codes.startsWith(hnode.getCode())) continue;
            if (isLeaf(node)) {
                sb.append(hnode.getName());
                codes = codes.substring(hnode.getCode().length());
                if (codes.length() > 0) { q.clear(); q.add(root); continue; }
            }
            if (node.getLchild() != null) { q.add(node.getLchild()); }
            if (node.getRchild() != null) { q.add(node.getRchild()); }
        }
        return codes.length() > 0 ? "" : sb.toString();
    }

    // ---- 辅助：对 Map 按 value 和 key 升序排序 ----
    private List<Map.Entry<Character, Integer>> sortMapByValueKey(
            Map<Character, Integer> map) {
        List<Map.Entry<Character, Integer>> list =
                new ArrayList<>(map.entrySet());
        Collections.sort(list, (a, b) -> {
            int cmp = a.getValue().compareTo(b.getValue());
            if (cmp != 0) return cmp;
            return a.getKey().compareTo(b.getKey());
        });
        return list;
    }
}
✅ 关键理解：LinkedHuffmanTree 使用 PriorityQueue（最小堆）自动维护结点优先级，建树过程只需要不断 poll 两个最小结点、合并、offer 回去，时间复杂度 O(n log n)。与 SeqHuffmanTree 的 select 线性扫描相比更高效。
